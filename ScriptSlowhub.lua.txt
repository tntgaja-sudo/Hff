-- NoClip bật tắt xuyên tường
AddToggle(Tab8, {
    Name = "NoClip",
    Default = false,
    Callback = function(State)
        local lp = game.Players.LocalPlayer
        getgenv().NoClip = State
        game:GetService("RunService").Stepped:Connect(function()
            if getgenv().NoClip and lp.Character then
                for _,v in pairs(lp.Character:GetDescendants()) do
                    if v:IsA("BasePart") then
                        v.CanCollide = false
                    end
                end
            end
        end)
    end
})

-- Fly bật/tắt bay
AddToggle(Tab8, {
    Name = "Fly",
    Default = false,
    Callback = function(State)
        local lp = game.Players.LocalPlayer
        local hrp = lp.Character:WaitForChild("HumanoidRootPart")
        getgenv().Flying = State
        if State then
            task.spawn(function()
                while getgenv().Flying and task.wait() do
                    hrp.Velocity = Vector3.new(0,50,0) -- bay lên
                end
            end)
        end
    end
})

-- Reset Character
AddButton(Tab8, {
    Name = "Reset Character",
    Callback = function()
        local lp = game.Players.LocalPlayer
        lp.Character:BreakJoints()
    end
})
