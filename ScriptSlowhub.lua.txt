local AutoFarm = false

-- Auto Farm Toggle
AddToggle(Tab1, {
    Name = "Auto Farm (All Levels)",
    Default = false,
    Callback = function(Value)
        AutoFarm = Value
        if AutoFarm then
            task.spawn(function()
                while AutoFarm do
                    pcall(function()
                        local player = game.Players.LocalPlayer
                        local character = player.Character
                        local hrp = character and character:FindFirstChild("HumanoidRootPart")
                        local level = player.Data.Level.Value
                        
                        if not hrp then return end

                        -- Tìm NPC nhiệm vụ đúng với level
                        for _, questNPC in pairs(workspace.NPCs:GetChildren()) do
                            if questNPC:FindFirstChild("LevelReq") 
                               and level >= questNPC.LevelReq.Value 
                               and level <= questNPC.MaxLevel.Value then
                                -- Tele tới NPC nhận nhiệm vụ
                                hrp.CFrame = questNPC.HumanoidRootPart.CFrame * CFrame.new(0,0,5)
                                fireproximityprompt(questNPC.ProximityPrompt) -- Nhận nhiệm vụ

                                -- Tìm quái của nhiệm vụ
                                for _, mob in pairs(workspace.Enemies:GetChildren()) do
                                    if mob:FindFirstChild("HumanoidRootPart") 
                                       and mob.Name == questNPC.TargetMob.Value then
                                        local mobHRP = mob.HumanoidRootPart
                                        -- Tele cách quái 10 studs rồi tiến tới
                                        hrp.CFrame = mobHRP.CFrame * CFrame.new(0,0,10)
                                        hrp.CFrame = hrp.CFrame:Lerp(mobHRP.CFrame * CFrame.new(0,0,3), 0.2)
                                        
                                        -- Tấn công
                                        if mob:FindFirstChild("Humanoid") then
                                            mob.Humanoid.Health = 0
                                        end
                                        break
                                    end
                                end
                            end
                        end
                    end)
                    task.wait(0.3)
                end
            end)
        end
    end
})
